```python
import functools, time as _time
_ttl_store: dict = {}
def ttl_cache(seconds: float):
    """Per-process TTL cache for read-only JSON endpoints. Key includes the
    request realm (contextvar) so realm-scoped reads never cross-serve."""
    def deco(fn):
        @functools.wraps(fn)
        async def wrapper(*args, **kwargs):
            key = (fn.__name__, args, tuple(sorted(kwargs.items())), _current_realm.get())
            hit = _ttl_store.get(key)
            now = _time.monotonic()
            if hit is not None and now - hit[0] < seconds:
                return hit[1]
            if len(_ttl_store) > 512:
                for k in [k for k, v in _ttl_store.items() if now - v[0] >= seconds]:
                    _ttl_store.pop(k, None)
            val = await fn(*args, **kwargs)
            _ttl_store[key] = (now, val)
            return val
        return wrapper
    return deco
```
